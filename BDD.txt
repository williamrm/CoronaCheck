PERSONNES (IDPers, Nom, Prenom, DateNaiss, Sexe)
TYPE_STATUT(idStatut, LibelleStatut)
VACCIN (Reference, DateInjection, DateRappel)
TYPE_VACCIN (idTypeVaccin, LibelleTypeVaccin)

CREATE TABLE PERSONNES (
	IDPers INTEGER,
	Nom varchar(50),
	Prenom varchar(50),
	DateNaiss date,
	Sexe boolean,
	idStatut INTEGER,
	PRIMARY KEY (IDPers)
	FOREIGN KEY (idStatut) REFERENCES TYPE_STATUT (idStatut)
)

CREATE TABLE TYPE_STATUT (
	idStatut INTEGER,
	LibelleStatut varchar(50),
	PRIMARY KEY (idStatut),

CREATE TABLE VACCIN (
	Reference INTEGER,
	DateInjection date,
	DateRappel date,
	idTypeVaccin INTEGER,
	PRIMARY KEY (Reference),
	FOREIGN KEY (idTypeVaccin) REFERENCES TYPE_VACCIN (idTypeVaccin)
)

CREATE TABLE TYPE_VACCIN (
	idTypeVaccin INTEGER,
	LibelleTypeVacci varchar(50),
	PRIMARY KEY (idTypeVaccin)
)
